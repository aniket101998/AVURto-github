    <input type="number" id="costInput" placeholder="Cost">
    <input type="number" id="gstInput" placeholder="GST (%)">
    <button onclick="calculate()">Calculate</button>
    <br>
    <label for="salePrice">Sale Price:</label>
    <span id="salePrice"></span>
</div>

<script>
    // Function to calculate sale price
    function calculate() {
        var cost = parseFloat(document.getElementById('costInput').value);
        var gst = parseFloat(document.getElementById('gstInput').value);

        if (!isNaN(cost) && !isNaN(gst)) {
            var salePrice = cost * (1 + gst / 100);
            document.getElementById('salePrice').innerText = salePrice.toFixed(2);
        } else {
            alert('Please enter valid numbers for Cost and GST.');
        }
    }
</script>
