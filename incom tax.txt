<script>
    document.getElementById('taxForm').addEventListener('submit', function(e) {
        e.preventDefault();
        var status = document.getElementById('status').value;
        var income = parseFloat(document.getElementById('income').value);
        var taxDetails = calculateTax(income, status);
        displayTaxDetails(taxDetails);
    });

    function calculateTax(income, status) {
        var taxSlabs;
        if (status === 'individual') {
            taxSlabs = [
                { min: 0, max: 250000, rate: 0 },
                { min: 250001, max: 500000, rate: 0.05 },
                { min: 500001, max: 1000000, rate: 0.2 },
                { min: 1000001, max: Infinity, rate: 0.3 }
            ];
        } else if (status === 'senior') {
            taxSlabs = [
                { min: 0, max: 300000, rate: 0 },
                { min: 300001, max: 500000, rate: 0.05 },
                { min: 500001, max: 1000000, rate: 0.2 },
                { min: 1000001, max: Infinity, rate: 0.3 }
            ];
        } else if (status === 'superSenior') {
            taxSlabs = [
                { min: 0, max: 500000, rate: 0 },
                { min: 500001, max: 1000000, rate: 0.05 },
                { min: 1000001, max: Infinity, rate: 0.3 }
            ];
        }

        var tax = 0;
        var remainingIncome = income;

        taxSlabs.forEach(function(slabs) {
            var taxableIncome = Math.min(remainingIncome, slabs.max - slabs.min);
            tax += taxableIncome * slabs.rate;
            remainingIncome -= taxableIncome;
        });

        return {
            income: income.toFixed(2),
            incomeTax: tax.toFixed(2),
            surcharge: (tax * 0.1).toFixed(2), // Surcharge at 10% of tax
            cess: (tax * 0.04).toFixed(2), // Health and Education cess at 4% of tax
            totalTax: (tax + tax * 0.1 + tax * 0.04).toFixed(2) // Total tax liability
        };
    }

    function displayTaxDetails(taxDetails) {
        document.getElementById('incomeDisplay').textContent = taxDetails.income + " INR";
        document.getElementById('incomeTax').textContent = taxDetails.incomeTax + " INR";
        document.getElementById('surcharge').textContent = taxDetails.surcharge + " INR";
        document.getElementById('cess').textContent = taxDetails.cess + " INR";
        document.getElementById('totalTax').textContent = taxDetails.totalTax + " INR";
        document.getElementById('taxDetails').style.display = 'table';
    }
</script>
