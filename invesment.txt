<script>
    // Function to calculate investment return
    function calculateReturn() {
        var initialInvestment = parseFloat(document.getElementById('initialInvestment').value);
        var finalValue = parseFloat(document.getElementById('finalValue').value);
        var investmentPeriod = parseFloat(document.getElementById('investmentPeriod').value);

        // Calculate net return
        var netReturn = finalValue - initialInvestment;

        // Calculate annualized return
        var annualizedReturn = ((Math.pow((finalValue / initialInvestment), (1 / investmentPeriod)) - 1) * 100).toFixed(2);

        // Display calculated values
        document.getElementById('netReturn').innerText = netReturn.toFixed(2);
        document.getElementById('annualizedReturn').innerText = annualizedReturn;
    }
</script>
